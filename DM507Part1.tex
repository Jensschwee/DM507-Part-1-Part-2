     * with the lowest key.
     */
    @Override

        Element min;
        min = heap[0];
        heap[0] = heap[heapsize-1];
        heapsize--;
        minheapify(0);
        return min;
    }

	public boolean search(int k) {
        Node x = root;
        while (x != null && k != x.key) {
            if (k < x.key) {
                x = x.left;
            } else {
                x = x.right;
            }
        }

        if (x == null) {
            return false;
        } else if (x.key == k) {
            return true;
        }
        return false;
    }
\end{lstlisting}
